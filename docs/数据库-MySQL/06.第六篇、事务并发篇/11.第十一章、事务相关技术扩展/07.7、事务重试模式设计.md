    private final RetryStateManager stateManager;

    // æ‰§è¡Œå¸¦é‡è¯•çš„äº‹åŠ¡
    public <T> T executeWithRetry(String txId, RetryableTransaction<T> transaction) {

        try {
            return doExecuteWithRetry(state, transaction);

            monitor.recordRetryEnd(sample, state.isSuccess(), 
                                 state.getLastErrorType(), state.getRetryCount());
            stateManager.clearState(txId);

    private <T> T doExecuteWithRetry(RetryState state, 
                                   RetryableTransaction<T> transaction) {
        while (true) {
            try {
                // æ‰§è¡Œäº‹åŠ¡
                T result = transaction.execute();
                state.markSuccess();
                return result;
                
            } catch (Exception ex) {
