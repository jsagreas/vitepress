## 一、导数运算法则表

  $$\begin{cases}
  & \text{} [f(x)+g(x)]'=f'(x)+g'(x)   \\  \newline
  & \text{} [f(x)-g(x)]'=f'(x)-g'(x)   \\  \newline
  & \text{} [f(x)⋅g(x)]'=f'(x)g(x)+f(x)g'(x)   \\  \newline
  & \text{} [\dfrac{f(x)}{g(x)}]'=\dfrac{ f'(x)g(x)−f(x)g'(x)}{[g(x)]^2} \quad (g(x)≠0)   \\  \newline
  & \text{} F'(u)=\dfrac{d y}{d x}=\dfrac{d y}{d u} \cdot \dfrac{d u}{d x}=f^{\prime}(u) \cdot g^{\prime}(x)   \quad (u=g(x),y=f(u))  \\  
  \end{cases}$$


## 二、导数运算法则推导


- 加法求导法则推导:

$$ \begin{cases}
&  f'(x)=\lim_{x \to 0} \dfrac{f(x+h)-f(x)}{h}\\  \newline
&  [ f(x)+g(x)]'=\lim_{h \to 0} \dfrac{[f(x+h)+g(x+h)]-[f(x)+g(x)]}{h}\\ \newline
&  =\lim_{h \to 0} \dfrac{[f(x+h)-f(x)]+[g(x+h)-g(x) ]}{h}\\  \newline
&  =f'(x)+g'(x)\\ 
\end{cases}$$



- 减法求导法则推导: 

$$  \begin{cases}
& f'(x)=\lim_{x \to 0} \dfrac{f(x+h)-f(x)}{h}\\  \newline
& [ f(x)-g(x)]'=\lim_{h \to 0} \dfrac{[f(x+h)-g(x+h)]-[f(x)-g(x)]}\\ \newline
& =\lim_{h \to 0} \dfrac{[f(x+h)-f(x)]-[g(x+h)-g(x) ]}{h}\\ \newline
& =f'(x)-g'(x)\\ 
\end{cases}$$

- 乘法求导法则推导:
$$ {\begin {cases} 
& f'(x)=\lim_{x \to 0} \dfrac{f(x+h)-f(x)}{h}\\  \newline
& [f(x)·g(x)]'=\lim_{h \to 0}\dfrac{[f(x+h)·g(x+h)]-[f(x)·g(x)]}{h} \\ \newline
& =\lim_{h \to 0}\dfrac{[f(x+h)-f(x)+f(x)]·[g(x+h)-g(x)+g(x)]-[f(x)·g(x)]}{h} \\ \newline
& 令:\quad f(x+h)-f(x)=a, g(x+h)-g(x)=b\\ \newline
& =\lim_{h \to 0}\dfrac{[a+f(x)]·[b+g(x)]- [f(x)·g(x)]}{h} \\ \newline
& =\lim_{h \to 0}\dfrac{[ab+ag(x)+bf(x)+f(x)g(x)]- [f(x)·g(x)]}{h}=\lim_{h \to 0}\dfrac{ab+ag(x)+bf(x)}{h} \\ \newline
& =\lim_{h \to 0}\dfrac{ab+ag(x)+bf(x)}{h}  \quad h→0 ⟹ ab=0 \\ \newline
& =\lim_{h \to 0}\dfrac{ag(x)+bf(x)}{h}=\dfrac{[f(x+h)-f(x)]·g(x)+[g(x+h)-g(x)·f(x)]}{h} \\ \newline
& =\lim_{h \to 0}\dfrac{f(x+h)-f(x)}{h}·g(x)+\dfrac{g(x+h)-g(x)}{h}·f(x) \\ \newline
& =\lim_{h \to 0}f'(x)g(x)+f(x)g'(x)
\end{cases}}$$


- 除法求导法则推导:

$$ {\begin {cases} 
& f'(x)=\lim_{x \to 0} \dfrac{f(x+h)-f(x)}{h}\\  \newline
& [\dfrac{f(x)}{g(x)}]'=\lim_{h \to 0}\dfrac{\dfrac{f(x+h)}{g(x+h)}-\dfrac{f(x)}{g(x)}}{h}\\ \newline
& =\lim_{h \to 0}\dfrac{f(x+h)g(x)-f(x)g(x+h)}{g(x)g(x+h)h}\\ \newline
& =\lim_{h \to 0}\dfrac{[f(x+h)-f(x)+f(x)]·g(x)-f(x)·[g(x+h)-g(x)+g(x)]}{g(x)g(x+h)h} \\ \newline
& 令:\quad f(x+h)-f(x)=a, g(x+h)-g(x)=b\\ \newline
& =\lim_{h \to 0}\dfrac{[a+f(x)]·g(x)-f(x)·[b+g(x)]}{g(x)g(x+h)h} \\ \newline
& =\lim_{h \to 0}\dfrac{ag(x)+f(x)g(x)-bf(x)-f(x)g(x)}{g(x)g(x+h)h}=\lim_{h \to 0}\dfrac{ag(x)-bf(x)}{g(x)g(x+h)h} \\ \newline
& =\lim_{h \to 0}\dfrac{ag(x)}{g(x)g(x+h)h}-\dfrac{bf(x)}{g(x)g(x+h)h} \\ \newline
& =\lim_{h \to 0}\dfrac{[f(x+h)-f(x)]·g(x)}{h·g(x)g(x+h)}-\dfrac{[g(x+h)-g(x)]·f(x)}{h·g(x)g(x+h)} \\ \newline
& =\lim_{h \to 0}\dfrac{f'(x)g(x)}{g(x)g(x+h)}-\dfrac{g'(x)f(x)}{g(x)g(x+h)} \\ \newline
& 当\quad h→0 ⟹ g(x+h)=g(x)\\ \newline
& =\lim_{h \to 0}\dfrac{f'(x)g(x)-g'(x)f(x)}{[g(x)]^2}
\end{cases}}$$


- 复合函数求导:等于外层函数对内层函数的导数乘以内层函数对自变量$x$的导数(链式法则)

$$ {\begin {cases} 
&  设y=f(g(x))，其中u=g(x)，则y=f(u) \\  \newline
&   y 对 x 的导数为：\frac{d y}{d x}=\lim _{\Delta x \rightarrow 0} \frac{\Delta y}{\Delta x}   \\  \newline
&   其中，\Delta y=f(g(x+\Delta x))-f(g(x))    \\  \newline
&   令\Delta u=g(x+\Delta x)-g(x),则：  \Delta y=f(u+\Delta u)-f(u)        \\  \newline
&   因此，导数可以表示为：  \frac{d y}{d x}=\lim _{\Delta x \rightarrow 0} \frac{f(u+\Delta u)-f(u)}{\Delta x}        \\  \newline
&   将极限分解为两部分：\frac{d y}{d x}=\lim _{\Delta x \rightarrow 0}\left(\frac{f(u+\Delta u)-f(u)}{\Delta u} \cdot \frac{\Delta u}{\Delta x}\right)   \\  \newline
&  当 \Delta x \rightarrow 0 时，\Delta u \rightarrow 0，因此：\frac{d y}{d x}=\left(\lim _{\Delta u \rightarrow 0} \frac{f(u+\Delta u)-f(u)}{\Delta u}\right) \cdot\left(\lim _{\Delta x \rightarrow 0} \frac{\Delta u}{\Delta x}\right)    \\  \newline
&  \lim _{\Delta u \rightarrow 0} \frac{f(u+\Delta u)-f(u)}{\Delta u}=\frac{d f}{d u}  \quad    \lim _{\Delta x \rightarrow 0} \frac{\Delta u}{\Delta x}=\frac{d u}{d x}    \\  \newline
&   因此，复合函数的导数为：\frac{d y}{d x}=\frac{d f}{d u} \cdot \frac{d u}{d x}=  \frac{d y}{d x}=f^{\prime}(u) \cdot g^{\prime}(x)        \\  \newline
&  将u=g(x) 代入，得到最终的复合函数求导公式： \frac{d y}{d x}=f^{\prime}(g(x)) \cdot g^{\prime}(x)    \\  
\end{cases}}$$

## 四、常用导数公式推导过程


- $1.$常数函数:$C^{\prime}=0  \quad (C$为常数)

$$C'=0 \quad {\begin{cases}
 & C'=0        \\ \newline
 & \lim_{h \to 0} \frac{f(x+h)-f(x)}{h}  \\ \newline
 & \lim_{h \to 0} \frac{C-C}{h}  \\ \newline
 & \lim_{h \to 0} \frac{0}{h}=0  
\end{cases}}$$


- $2.$幂函数: $(x^n)'=nx^{n−1} \quad (n∈Q$,且$n≠0)$
$$ (x^n)'=nx^{n−1} \quad{\begin{cases}
 & \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{(x+h)^n-x^n}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{(C_n^0x^nh^0+C_n^1x^{n-1}h^1+C_n^2x^{n-2}h^2+...+c_n^nh^n)-x^n}{h}\\ \newline
 & \lim_{h \to 0} \dfrac{(x^n+nx^{n-1}h+C_n^2x^{n-2}h^2+...+h^n)-x^n}{h} \\ \newline
 & \lim_{h \to 0} \dfrac{x^n-x^n}{h}+ nx^{n-1}+ C_n^2x^{n-2}h+...+ h^{n-1} \\ \newline
 &  h→0 ⟹ h=h^k=0 \\ \newline
 &\lim_{h \to 0} (x^n)'=nx^{n−1}
\end{cases}}$$


- $3.$指数函数特例：$(e^x)'=e^x$


- $4.$指数函数:$(a^x)'=a^xlna \quad (a>0$,且$a≠1)$
$$ (a^x)'=a^xlna \quad {\begin{cases}
 &  \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}\\ \newline
 &  \lim_{h \to 0} \dfrac{a^{x+h}-a^x}{h}\\ \newline
 & 令t=lna \Rightarrow a^h=e^{th}   \\ \newline
 & 令u=th \quad a^h-1 \Rightarrow   e^u-1  \\ \newline

  & h \to 0, u \to 0 \quad    e^u-1 \sim u \Rightarrow h \cdot  lna  \\ \newline
 &  \lim_{h \to 0} \dfrac{a^x(a^h-1)}{h}\\  \newline
 &  \lim_{h \to 0} \dfrac{a^x·hlna}{h}=a^xlna\\  \newline
 &  特别的,当a=e \quad 则有e^xlne=e^x
\end{cases}}$$


- $5.$对数函数特例：$(lnx)'=\frac{1}{x} \quad (x>0)$


- $6.$对数函数: $(log_ax)'=\large{\frac{1}{xlna}} \quad (x>0,a>0,且a≠1)$

$$ (log_ax)'=\dfrac{1}{xlna}\quad {\begin{cases}
 & \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{log_a(x+h)-log_ax}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{log_a(\frac{x+h}{x})}{h} =\lim_{h \to 0} \frac{log_a(1+\frac{h}{x})}{h} \\ \newline
 & \lim_{h \to 0} log_a(1+\frac{h}{x})·\frac{1}{h}  \\ \newline
 & \lim_{h \to 0} log_a(1+\frac{h}{x})^{\frac{1}{h}}  \\ \newline
 & \lim_{h \to 0} log_a[(1+\frac{h}{x})^{\frac{x}{h}}]^{\frac{1}{x}}  \\ \newline
 & \lim_{h \to 0} log_ae^{\frac{1}{x}} =\frac{1}{x}log_ae \\ \newline
 & \lim_{h \to 0} \dfrac{1}{x}log_ae =\frac{1}{x}·\frac{lne}{lna} \\ \newline
& \lim_{h \to 0} \dfrac{1}{x}log_ae =\frac{1}{xlna}  \\ \newline
&  特别的,当a=e则有\dfrac{1}{xlna}=\frac{1}{x}
\end{cases}}$$



 - $7.$正弦函数：$(sinx)'=cosx$
$$ (sinx)'=cosx  \quad{\begin{cases}
 & \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{sin(x+h)-sinx}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{sinxcosh+cosxsinh-sinx}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{sinxcos0+cosxsinh-sinx}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{sinx+cosxsinh-sinx}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{cosxsinh}{h}  \\ \newline
 & \lim_{h \to 0} =cosx
\end{cases}}$$



 - $8.$余弦函数： $(cosx)'=−sinx$

$$(cosx)'=−sinx  \quad{\begin{cases}
 & \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{cos(x+h)-cos x}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{cosxcosh-sinxsinh-cosx}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{cosxcos0-sinxsinh-cosx}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{cosx-sinxsinh-cosx}{h}  \\ \newline
 & \lim_{h \to 0} \dfrac{-sinxsinh}{h}  \\ \newline
 & \lim_{h \to 0} =-sinx  \\
\end{cases}}$$


  - $9.$正切函数：$(tanx)'=\large{\frac{1}{ cos^2x} }=sec^2x$
$$(tanx)'=\frac{1}{ cos^2x}=sec^2x \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& \lim_{h \to 0} (\dfrac{sinx}{cosx})'  \\ \newline
& \lim_{h \to 0} \dfrac{cosx·cosx-sinx·(-sinx)}{cos^2x}  \\ \newline
& \lim_{h \to 0} \dfrac{cos^2x+sin^2x}{cos^2x}  \\ \newline
& \lim_{h \to 0} \dfrac{1}{cos^2x}=sec^2x 
\end{cases}}$$


  - $10.$余切函数$(cotx)'=\large{−\frac{1}{sin^2x}}=−csc^2x$
$$(cotx)'=−\dfrac{1}{sin^2x}=−csc^2x \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& \lim_{h \to 0} (\dfrac{cos}{sinx})'  \\ \newline
& \lim_{h \to 0} \dfrac{-sinx·sinx-cosx·cosx}{sin^2x}  \\ \newline
& \lim_{h \to 0} \dfrac{-sin^2x-cos^2x}{sin^2x}  \\ \newline
& \lim_{h \to 0} \dfrac{-1}{sin^2x}  \\ \newline
& \lim_{h \to 0} =-csc^2x 
\end{cases}}$$


  - $11.$正割函数$(secx)'=secxtanx$
$$(secx)'=secxtanx  \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& \lim_{h \to 0} (\dfrac{1}{cosx})'  \\ \newline
& \lim_{h \to 0} \dfrac{0·cosx-1·(-sinx)}{cos^2x} \\ \newline
& \lim_{h \to 0} \dfrac{sinx}{cos^2x} \\ \newline
& \lim_{h \to 0} \dfrac{1}{cosx} ·\dfrac{sinx}{cosx} \\ \newline
& \lim_{h \to 0} =secxtanx 
\end{cases}}$$


 - $12.$余割函数$(cscx)'=-cscxcotx$

$$(cscx)'=-cscxcotx \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& \lim_{h \to 0} (\dfrac{1}{sinx})'  \\ \newline
& \lim_{h \to 0} \dfrac{0·sinx-1·cosx}{sin^2x} \\ \newline
& \lim_{h \to 0} \dfrac{-cosx}{sin^2x} \\ \newline
& \lim_{h \to 0} -\dfrac{1}{sinx} ·\dfrac{cosx}{sinx} \\ \newline
& \lim_{h \to 0} =-cscxcotx 
\end{cases}}$$


 - $13.$反正弦函数$(arcsinx)'= \frac{1}{\sqrt{1-x^2}}$

$$(arcsinx)'= \dfrac{1}{\sqrt{1-x^2}}  \quad{\begin{cases} 
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline 
& f(x)=sinx⟹ y=f^{-1}(x)=arcsinx      \\ \newline 
& [ f^{-1}(x)]' =\dfrac{1}{f'(f^{-1}(x))}=\dfrac{1}{f'(y)}\\ \newline 
& =\dfrac{1}{cosy}=\dfrac{1}{\sqrt{1-sin^2y}}\\ \newline 
& =(arcsinx)'=\dfrac{1}{\sqrt{1-x^2}} 
\end{cases}}$$ 



  - $14.$反余弦函数$(arccosx)'=  - \frac{1}{\sqrt{1-x^2}}$
$$(arccosx)'=  - \dfrac{1}{\sqrt{1-x^2}} \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& f(x)=cosx⟹y=arccosx=f^{-1}(x)\\ \newline
& [f^{-1}(x)]'=\dfrac{1}{f'(f^{-1}(x))}=\dfrac{1}{f'(y)}\\ \newline
& =\dfrac{1}{-siny}=\dfrac{1}{-\sqrt{1-cos^2y}}\\ \newline
& =(arccosx)'=-\dfrac{1}{\sqrt{1-x^2}}
\end{cases}}$$


  - $15.$反正切函数$(arctanx)'=  \frac{1}{1+x^2}$
$$(arctanx)'=  \dfrac{1}{1+x^2} \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& f(x)=tanx⟹f'(x)=sec^2x\\ \newline
& sec^2x=1+tan^2x \\ \newline
& y=arctanx=f^{-1}(x)\\ \newline
& [f^{-1}(x)]'=\dfrac{1}{f'(f^{-1}(x))}=\dfrac{1}{f'(y)}\\ \newline
& =\dfrac{1}{sec^2y}=\dfrac{1}{1+tan^2y}\\ \newline
& =(arctanx)'=\dfrac{1}{1+x^2} \\
\end{cases}}$$


  - $16.$反余切函数$(arccotx)'= - \frac{1}{1+x^2}$
$$(arccotx)'= - \dfrac{1}{1+x^2} \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& f(x)=cotx⟹f'(x)=-csc^2x\\ \newline
& csc^2x=1+cot^2x \\ \newline
& y=arccotx=f^{-1}(x)\\ \newline
& [f^{-1}(x)]'=\dfrac{1}{f'(f^{-1}(x))}=\dfrac{1}{f'(y)}\\ \newline
& =\dfrac{1}{-csc^2y}=\dfrac{1}{-(1+cot^2x)}\\ \newline
& =(arctanx)'=-\dfrac{1}{1+x^2} 
\end{cases}}$$


 - $17.$反正割函数$(arcsecx)'=\dfrac{1}{|x|\sqrt{x^2-1}}$
$$(arcsecx)'=\dfrac{1}{|x|\sqrt{x^2-1}}  \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& f(x)=secx⟹f'(x)=secx·tanx  \\ \newline
& sec^2x=1+tan^2x ⟹ tanx =\pm  \sqrt{sec^2x-1}    \\ \newline
& y=arcsecx=f^{-1}(x)\\ \newline
& [f^{-1}(x)]'=\dfrac{1}{f'(f^{-1}(x))}=\dfrac{1}{f'(y)}\\ \newline
& 设符号函数为 \quad {sgn}(x)=\dfrac{x}{|x|}  \\ \newline
& =\dfrac{1}{secy·tany}=\dfrac{1}{secy  \cdot  (\dfrac{x}{|x|}  \cdot \sqrt{sec^2y-1})}\\ \newline
& =(arcsecx)'= \frac{1}{x \cdot \dfrac{x}{|x|} \cdot \sqrt{x^{2}-1}}=\dfrac{1}{|x| \cdot \sqrt{x^{2}-1}} \\ 
\end{cases}}$$


 - $18.$反余割函数$(arccscx)'= - \dfrac{1}{|x|\sqrt{x^2-1}}$
$$(arccscx)'= - \dfrac{1}{|x|\sqrt{x^2-1}} \quad{\begin{cases}
& \lim_{h \to 0} \dfrac{f(x+h)-f(x)}{h}  \\ \newline
& f(x)=cscx⟹f'(x)=-cscx·cotx\\ \newline
& csc^2x=1+cot^2x  ⟹ cotx=\pm  \sqrt{csc^2x-1}  \\ \newline
& y=arccscx=f^{-1}(x)\\ \newline
& [f^{-1}(x)]'=\dfrac{1}{f'(f^{-1}(x))}=\dfrac{1}{f'(y)}\\ \newline

&设符号函数为 \quad {sgn}(x)=\dfrac{x}{|x|}  \\ \newline
& =\dfrac{1}{-cscy·coty}=\dfrac{1}{-cscy \cdot (\dfrac{x}{|x|}  \cdot \sqrt{csc^2y-1})}\\ \newline
& =(arcsecx)'=\frac{1}{-x \cdot \frac{x}{|x|} \cdot \sqrt{x^{2}-1}}=-\frac{1}{|x| \cdot \sqrt{x^{2}-1}}  \\ 
\end{cases}}$$


## 五、常见函数导数值


$${\begin{cases}
 &C'=0  \quad (C为常数)  \\ \newline
 &(x)'=1        \\ \newline
 &(x^2)'=2x        \\ \newline
 &(x^3)'=3x^2        \\ \newline
 &(\frac{1}{x})'=-\frac{1}{x^2}    \\ \newline
 &(cf(x))'=cf'(x) \quad (c为常数)  \\ \newline
 &(\dfrac{1}{f(x)})'=−\dfrac{f'(x)}{[f(x)]^2}
\end{cases}}$$
